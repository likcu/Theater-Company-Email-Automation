39    @Test
40    public void checkArguments() {
41      String example = "Usage:\n\n"
42          + "        --email                        only generate email messages\n"
43          + "        --email-template <file>  accepts a filename that holds the email "
44          + "template.\n"
45          + "          Required if --email is used\n\n"
46          + "        --letter                        only generate letters\n"
47          + "        --letter-template <file> accepts a filename that holds the email "
48          + "template.\n"
49          + "          Required if --letter is used\n\n"
50          + "        --output-dir <path> accepts the name of a folder, all output is placed "
51          + "in this folder\n"
52          + "\n"
53          + "        --csv-file <path> accepts the name of the csv file to process\n";
54      String actionError = "Error: no action provided.\n";
55      String templateError = "Error: --email provided but no --email-template was given.\n";
56      String dirError = "Error: output directory was not given.\n";
57      String inputError = "Error: member information file was not given.\n";
58      String[] noAction = {"--email-template", "email-template.txt", "--output-dir", "emails",
59          "--csv-file", "customer.csv"};
60      String[] noTemplate1 = {"--email", "--email-template", "--output-dir",
61          "emails", "--csv-file", "customer.csv"};
62      ParseArgument parseTest3 = new ParseArgument(noAction);
63      ParseArgument parseTest4 = new ParseArgument(noTemplate1);
64      Assert.assertEquals("", parseTest1.checkArguments());
65      Assert.assertEquals("", parseTest2.checkArguments());
66      Assert.assertEquals(actionError + example, parseTest3.checkArguments());
67      Assert.assertEquals(templateError + example, parseTest4.checkArguments());
68    }
