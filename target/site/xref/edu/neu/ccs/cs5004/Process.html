3   import java.util.ArrayList;
4   import java.util.List;
5   import java.util.Map;

7   public class Process {
8     private Member member;
9     private List<String> template;
10  
11    /**
12     * The constructor.
13     * @param member The member
14     * @param template The template file string list
15     */
16    public Process(Member member, List<String> template) {
17      this.member = member;
18      this.template = template;
19    }
20  
21    /**
22     * Generate the normal file with a person info.
23     * @param onePerson One person info from the member
24     * @return The normal file list of strings
25     */
26    public List<String> generate(String[] onePerson) {
27      List<String> result = new ArrayList<>();
28      Map<String, Integer> map = member.getMap();
29      for (String str : template) {
30        int index = str.indexOf("[[");
31        while (index != -1) {
32          int end = str.indexOf("]]", index);
33          String header = str.substring(index + 2, end);
34          str = str.replaceAll("\\[\\[" + header + "\\]\\]", onePerson[map.get(header)]);
35          index = str.indexOf("[[");
36        }
37        result.add(str);
38      }
39      return result;
40    }
